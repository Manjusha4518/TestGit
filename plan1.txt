        return coverageCode.trim();
    }

    private boolean isBuyUp(String text) {
        if (text == null) return false;
        String s = text.trim().toUpperCase();
        return s.contains("STDBU") || s.contains("BUY UP");
    }


    private String shortLabel(String label, String prd) {
        if (label == null) return "";
        String s = label.trim();
        if (prd != null && s.toUpperCase().startsWith(prd.trim().toUpperCase() + " ")) {
            return s.substring(prd.length()).trim();
        }
        if (s.toUpperCase().startsWith("STD ")) return s.substring(4).trim();
        if (s.toUpperCase().startsWith("VPS ")) return s.substring(4).trim();
        return s;
    }

    private void openCoverageInUI(WebDriver driver, int planIndex, String prd, String coverageLabel) throws InterruptedException {

        By coverageLoc = By.xpath("//div[text()='" + prd + " Plan " + planIndex + "']/../../../following-sibling::div//div[text()='" + prd + " " + coverageLabel + "']");

        WebElement covLoc = driver.findElement(coverageLoc);

        ((JavascriptExecutor) driver).executeScript("arguments[0].scrollIntoView();", covLoc);
        WebDriverWait wait = new WebDriverWait(driver, Duration.ofSeconds(15));
        wait.until(ExpectedConditions.elementToBeClickable(coverageLoc));
        base.Common.Click(driver, covLoc, sc, "Open Coverage", true, false, true, false, 20, 2);
        Thread.sleep(1000);

    }
