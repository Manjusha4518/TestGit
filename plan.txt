    /** True if this string indicates a Buy Up (code or visible text). */
    private boolean isBuyUp(String text) {
        if (text == null) return false;
        String s = text.trim().toUpperCase();
        return s.contains("BUYUP") || s.contains("BUY UP");
    }

    /** Remove product prefix (e.g., "STD Core" -> "Core", "VPS Buy Up 1" -> "Buy Up 1"). */
    private String shortLabel(String label, String prd) {
        if (label == null) return "";
        String s = label.trim();
        if (prd != null && s.toUpperCase().startsWith(prd.trim().toUpperCase() + " ")) {
            return s.substring(prd.length()).trim();
        }
        if (s.toUpperCase().startsWith("STD ")) return s.substring(4).trim();
        if (s.toUpperCase().startsWith("VPS ")) return s.substring(4).trim();
        return s;
    }

    /** Nth plan card (1-based). No header text dependency. */
    private String planContainerByIndex(int planIndex) {
        return "(//div[contains(@class,'gen-plans-menu-item')])[" + planIndex + "]";
    }

    /** Check if a coverage exists under the nth plan card (accepts short or full label). */
    private boolean isCoveragePresentInUI(WebDriver driver, int planIndex, String prd, String coverageLabel) {
        String card = planContainerByIndex(planIndex);
        String shortLbl = shortLabel(coverageLabel, prd);
        String xp = card + "//*[self::div or self::span or self::button]" +
                "[normalize-space()='" + shortLbl + "' or normalize-space()='" + coverageLabel + "']";
        return !driver.findElements(By.xpath(xp)).isEmpty();
    }

    /** Always scroll into view, then click the coverage under nth card. */
    private void openCoverageInUI(WebDriver driver, int planIndex, String prd, String coverageLabel) {
        WebDriverWait wait = new WebDriverWait(driver, Duration.ofSeconds(15));
        String card = planContainerByIndex(planIndex);
        String shortLbl = shortLabel(coverageLabel, prd);
        By cov = By.xpath(card + "//following::div[normalize-space()='" + shortLbl + "' or normalize-space()='" + coverageLabel + "']");
        WebElement el = wait.until(ExpectedConditions.presenceOfElementLocated(cov));
        ((JavascriptExecutor) driver).executeScript("arguments[0].scrollIntoView({block:'center', inline:'nearest'});", el);
        wait.until(ExpectedConditions.elementToBeClickable(cov)).click();
    }

    /** Open selection modal via PLUS (global button). No plan header waiting. */
    private void openModalViaPlus(WebDriver driver) {
        WebDriverWait wait = new WebDriverWait(driver, Duration.ofSeconds(15));
        WebElement plus = wait.until(ExpectedConditions.visibilityOf(std.selectPlanPlusSign_Btn));
        ((JavascriptExecutor) driver).executeScript("arguments[0].scrollIntoView({block:'center'});", plus);
        try {
            wait.until(ExpectedConditions.elementToBeClickable(std.selectPlanPlusSign_Btn)).click();
        } catch (Exception e) {
            ((JavascriptExecutor) driver).executeScript("arguments[0].click();", std.selectPlanPlusSign_Btn);
        }
    }

    /** Open plan menu (three dots) for nth card and click "Add Coverage". */
    private void openMenuAndAddCoverage(WebDriver driver, int planIndex) throws InterruptedException {
        WebDriverWait wait = new WebDriverWait(driver, Duration.ofSeconds(15));
        String card = planContainerByIndex(planIndex);

        By menuBtn = By.xpath(card + "//div[contains(@class,'gen-plans-menu-item-icons')]");
        WebElement m = wait.until(ExpectedConditions.elementToBeClickable(menuBtn));
        ((JavascriptExecutor) driver).executeScript("arguments[0].scrollIntoView({block:'center'});", m);

        By addCov = By.xpath("//div[contains(@class,'ant-dropdown')]//li[normalize-space()='Add Coverage']");
        wait.until(ExpectedConditions.elementToBeClickable(addCov)).click();
    }
